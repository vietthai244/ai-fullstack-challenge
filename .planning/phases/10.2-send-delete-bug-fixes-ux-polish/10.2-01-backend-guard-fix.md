<objective>
Fix two runtime bugs that both share the same root cause: passing `{ updatedAt: new Date() }` to
`Campaign.update()` on a model with `timestamps: true`. Sequelize auto-manages `updatedAt` and
STRIPS it from the supplied values object, producing an empty SET clause. The resulting SQL
executes zero rows (`count === 0`), triggering the guard bail path even for legitimately valid
campaigns.

Fix: replace `{ updatedAt: new Date() }` with `{ status: <current-status> }` — an idempotent
no-op value change that forces Sequelize to emit a real SET clause (`status = 'sending'` or
`status = 'draft'`). `timestamps: true` then auto-appends `updated_at = NOW()`. The WHERE clause
`status = <same-value>` still acts as the atomic guard.

Purpose: Unblock the two primary UAT failures — Send button leaving campaign stuck forever in
"sending" state, and Delete returning 409 CAMPAIGN_NOT_EDITABLE on a valid draft.

Output: Two patched service files; backend typechecks; manual smoke confirms send completes and
delete succeeds.
</objective>

export async function deleteCampaign(campaignId: number, userId: number): Promise<void> {
  await sequelize.transaction(async (t) => {
    const campaign = await Campaign.findOne({
      where: { id: campaignId, createdBy: userId },
      transaction: t,
    });
    if (!campaign) throw new NotFoundError('CAMPAIGN_NOT_FOUND');

    // BUG (line 224-228): same updatedAt stripping issue
    const [count] = await Campaign.update(
      { updatedAt: new Date() },   // <-- BUGGY LINE
      { where: { id: campaignId, createdBy: userId, status: 'draft' }, transaction: t },
    );
    if (count === 0) throw new ConflictError('CAMPAIGN_NOT_EDITABLE');

    await Campaign.destroy({ where: { id: campaignId, createdBy: userId }, transaction: t });
  });
}
