The complete file must be:

```yaml
# docker-compose.yml (repo root)
# Phase 2: postgres. Phase 3 adds redis. Phase 10: api + web (full stack).
services:
  postgres:
    image: postgres:16-alpine
    environment:
      POSTGRES_USER: campaign
      POSTGRES_PASSWORD: campaign
      POSTGRES_DB: campaigns
    ports:
      - "5432:5432"
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U $$POSTGRES_USER -d $$POSTGRES_DB"]
      interval: 5s
      timeout: 5s
      retries: 10
    volumes:
      - pgdata:/var/lib/postgresql/data

  redis:
    image: redis:7-alpine
    ports:
      - "6379:6379"
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 5s
      timeout: 5s
      retries: 10
    volumes:
      - redisdata:/data

  api:
    build:
      context: .
      dockerfile: backend/Dockerfile
    env_file: .env
    environment:
      DATABASE_URL: postgres://campaign:campaign@postgres:5432/campaigns
      REDIS_URL: redis://redis:6379
      NODE_ENV: production
    depends_on:
      postgres:
        condition: service_healthy
      redis:
        condition: service_healthy

  web:
    build:
      context: .
      dockerfile: frontend/Dockerfile
    ports:
      - "8080:80"
    depends_on:
      api:
        condition: service_started

volumes:
  pgdata:
  redisdata:
```

Key decisions:
1. `api.env_file: .env` — loads all secrets (JWT_ACCESS_SECRET, JWT_REFRESH_SECRET, etc.) from .env at runtime
2. `api.environment` block overrides DATABASE_URL and REDIS_URL to use Docker service names (`postgres`, `redis`) — these override what's in .env which uses localhost
3. `api` has NO `ports:` binding — it is only reachable internally via `web` nginx proxy
4. `web.ports: 8080:80` — the only host-exposed port (FOUND-03)
5. `api.depends_on` uses `condition: service_healthy` for BOTH postgres and redis — prevents migration failures from race conditions (C14 in CLAUDE.md)
6. `web.depends_on` uses `condition: service_started` for api — api doesn't have a healthcheck but starts fast enough
